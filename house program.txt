function picketFences(x,y, range, col)
  pen col
  var r=0
  var px=0
  while r < range
      let px=x+r
      moveto px,y
      rectangle 10,70
      let r=r+20
  endloop
  var ly=y+10
  moveto x,ly
  rectangle 150,10
endfunction

function window(x,y,w,h,col)
  pen col
  moveto x,y
  rectangle w,h
  var mh=h/2 + y
  var dx=x+w
  var mw=w/2 + x
  var dy=y+h
  pen 'black'
  moveto x,mh
  drawto dx,mh
  moveto mw,y
  drawto mw, dy
endfunction

function beacon(x,y,w,h,flashcol,postcol)
  pen flashcol
  moveto x,y
  rectangle w,10
  pen postcol
  var py=y+10
  moveto x,py
  rectangle 10,100
endfunction
